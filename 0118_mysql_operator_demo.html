__#0118 ORDER BY ------------------------
used for ordering or sorting data

__#ex1 order by clause syntax
SELECT *
FROM employee
ORDER BY column_name ASC | DESC;


__#ex2 order by clause syntax > multiple columns and sort order
SELECT column1, column2
FROM table_name
ORDER BY column1 ASC, column2 DESC;


__#ex3
---------------sort by column ascending order
SELECT ID, first_name, last_name, nationality
FROM student_tbl
ORDER BY nationality ASC;


__#ex4
---------------sort by column decending order
SELECT ID, first_name, last_name, nationality
FROM student_tbl
ORDER BY nationality DESC;


__#ex5
---------------sort by column ascending order
SELECT ID, first_name, last_name, nationality
FROM student_tbl
ORDER BY nationality ASC, date_of_birth DESC;


__# WHERE clause ------------------------
used to filter and retrieve records that meet a specific condition

__# ex 0118
SELECT column1, column2
FROM table_name
WHERE student_id = 01;


__# ex2
SELECT column1, column2
FROM table_name
WHERE first_name = 'Jesus';


__# ex3

FROM student_tbl
WHERE faculty = 'Engineering';


__# ex4 using the between operator
-----------------------
SELECT *
FROM student_tbl
WHERE date_of_birth BETWEEN '2010-01-01' AND '2010-06-30';


__# ex5 using the like operator
-----------------------to find the faculty in science dept.
SELECT *
FROM student_tbl
WHERE faculty LIKE 'Sc%';


__# ex6 using the IN operator
-----------------------
SELECT *
FROM student_tbl

__# ex6 using the OR operator
-----------------------

FROM student_tbl
WHERE BillingCountry = 'USA' OR BillingCountry = 'France';

__# SELECT DISTINCT clause
---------------------returns distinct values, w/o duplicates
to filter out duplicates from the search add the distinct

__#ex0118
---------------
SELECT DISTINCT country
FROM student_tbl;


__#ex2 select distinct
--------------------
SELECT DISTINCT BillingCountry, BillingCity
FROM invoices
ORDER BY BillingCountry, BillingCity;


__#ex3 using count to count the total columns
output: 24------------------------------

FROM customers;
